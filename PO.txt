            static bool CzyLiczbaPierwsza(int liczba)
            {
                if (liczba <= 1)
                {
                    return false;
                }

                for (int i = 2; i * i <= liczba; i++)
                {
                    if (liczba % i == 0)
                    {
                        return false;
                    }
                }

                return true;
